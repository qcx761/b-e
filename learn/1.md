通用寄存器 ← 段寄存器 ✅ 合法

段寄存器的值可以读到通用寄存器里：

mov ax, cs   ; ✅ 合法
mov bx, ds   ; ✅ 合法
mov cx, ss   ; ✅ 合法
mov dx, es   ; ✅ 合法




2. 段寄存器 ← 通用寄存器 ✅ 合法

通用寄存器的值可以写进段寄存器：
mov ds, ax   ; ✅ 合法
mov es, bx   ; ✅ 合法
mov ss, cx   ; ✅ 合法
mov ds, dx   ; ✅ 合法





段寄存器 ← 段寄存器 ❌ 不合法
mov ds, cs   ; ❌ 非法
mov es, ds   ; ❌ 非法

mov ax, cs
mov ds, ax


4. 段寄存器 ← 立即数 ❌ 不合法
mov ds, 0x1234   ; ❌ 错误

mov ax, 0x1234
mov ds, ax
